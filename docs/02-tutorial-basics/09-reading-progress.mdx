# 🏄‍♂️ Tối Ưu Trải Nghiệm "Ăn Chơi Nhảy Múa" Cho User

Nhìn lên sát đỉnh màn hình của b đi! Một thanh tiến trình (Progress Bar) màu xanh sành điệu đang chạy mượt mà theo tốc độ cuộn chuột đọc bài của b đó. 

export const ReadingProgressBar = () => {
  const [scrollProgress, setScrollProgress] = useState(0);

  useEffect(() => {
    const handleScroll = () => {
      // Tính toán tổng độ cao có thể cuộn của trang tài liệu
      const totalHeight = document.documentElement.scrollHeight - document.documentElement.clientHeight;
      if (totalHeight > 0) {
        const progress = (window.scrollY / totalHeight) * 100;
        setScrollProgress(progress);
      }
    };

    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, []);

  return (
    <div style={{
      position: 'fixed',
      top: 'var(--ifm-navbar-height, 60px)', // Nằm ngay dưới thanh điều hướng của Docusaurus
      left: 0,
      width: `${scrollProgress}%`,
      height: '4px',
      background: 'linear-gradient(90deg, #2f6fe2, #00ced1, #2ecc71)', // Hiệu ứng dải màu gradient chuyển sắc mượt mà
      zIndex: 1000,
      transition: 'width 0.1s ease-out',
      boxShadow: '0 1px 10px rgba(47, 111, 226, 0.5)'
    }} />
  );
};

{/* Gài luôn CSS để độ lại nút Copy mặc định và hiệu ứng Block Code cho chất chơi */}
<style>{`
  /* 1. Tạo hiệu ứng viền phát sáng lơ lửng cho các khối Code Block */
  .theme-code-block {
    box-shadow: 0 4px 6px rgba(0, 0, 0, 0.05);
    transition: all 0.3s cubic-bezier(0.25, 0.8, 0.25, 1) !important;
    border: 1px solid transparent;
  }
  .theme-code-block:hover {
    transform: translateY(-2px);
    box-shadow: 0 12px 20px rgba(47, 111, 226, 0.15);
    border-color: rgba(47, 111, 226, 0.3);
  }

  /* 2. Độ lại nút Copy Code mặc định nhìn cho công nghệ */
  .copyButton_node_modules-\@docusaurus-theme-classic-src-theme-CodeBlock-CopyButton-styles-module {
    background-color: rgba(47, 111, 226, 0.1) !important;
    border: 1px solid rgba(47, 111, 226, 0.3) !important;
    border-radius: 6px !important;
    color: #2f6fe2 !important;
    transition: all 0.2s ease !important;
  }
  .copyButton_node_modules-\@docusaurus-theme-classic-src-theme-CodeBlock-CopyButton-styles-module:hover {
    background-color: #2f6fe2 !important;
    color: #fff !important;
    transform: scale(1.05);
  }
`}</style>

<BrowserOnly>
  {() => <ReadingProgressBar />}
</BrowserOnly>
